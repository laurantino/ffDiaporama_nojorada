848 struct LangAltValueComparator {
850 bool operator() (const std::string& str1, const std::string& str2) const
851 {
852 int result = str1.size() < str2.size() ? 1
853 : str1.size() > str2.size() ? -1
854 : 0
855 ;
856 std::string::const_iterator c1 = str1.begin();
857 std::string::const_iterator c2 = str2.begin();
858 if ( result==0 ) for (
859 ; result==0 && c1 != str1.end()
860 ; ++c1, ++c2
861 ) {
862 result = tolower(*c1) < tolower(*c2) ? 1
863 : tolower(*c1) > tolower(*c2) ? -1
864 : 0
865 ;
866 }
867 return result < 0 ;
868 }
869 };

1539 template<typename T>
1540 int ValueType<T>::read(const byte* buf, long len, ByteOrder byteOrder)
1541 {
1542 value_.clear();
1543 long ts = TypeInfo::typeSize(typeId());
1544 if (ts > 0)
1545 if (len % ts != 0) len = (len / ts) * ts;
1546 for (long i = 0; i < len; i += ts) {
1547 value_.push_back(getValue<T>(buf + i, byteOrder));
1548 }
1549 return 0;
1550 }

1552 template<typename T>
1553 int ValueType<T>::read(const std::string& buf)
1554 {
1555 std::istringstream is(buf);
1556 T tmp = T();
1557 ValueList val;
1558 while (!(is.eof())) {
1559 is >> tmp;
1560 if (is.fail()) return 1;
1561 val.push_back(tmp);
1562 }
1563 value_.swap(val);
1564 return 0;
1565 }

1596 template<typename T>
1597 std::ostream& ValueType<T>::write(std::ostream& os) const
1598 {
1599 typename ValueList::const_iterator end = value_.end();
1600 typename ValueList::const_iterator i = value_.begin();
