166 struct TiffImgTagStruct {
168 struct Key {
170 Key(uint16_t t, IfdId g) : t_(t), g_(g) {}
171 uint16_t t_;
172 IfdId g_;
173 };
174
176 bool operator==(const Key& key) const
177 {
178 return key.g_ == group_ && key.t_ == tag_;
179 }
180
181 // DATA
182 uint16_t tag_;
183 IfdId group_;
184 }; // struct TiffImgTagStruct

190 struct TiffGroupStruct {
192 struct Key {
194 Key(uint32_t e, IfdId g) : e_(e), g_(g) {}
195 uint32_t e_;
196 IfdId g_;
197 };
198
200 bool operator==(const Key& key) const
201 {
202 return key.g_ == group_
203 && (Tag::all == extendedTag_ || key.e_ == extendedTag_);
204 }
206 uint16_t tag() const { return static_cast<uint16_t>(extendedTag_ & 0xffff); }
207
208 // DATA
209 uint32_t extendedTag_;
210 IfdId group_;
211 NewTiffCompFct newTiffCompFct_;
212 };
